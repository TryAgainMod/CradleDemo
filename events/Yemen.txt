# Declares a civil war!
country_event = {
	id = yemen.2
	title = yemen.2.t
	desc = yemen.2.d
	picture = GFX_report_event_saf_civil_war

	is_triggered_only = yes	

	option = {
		name = yemen.2.a
		set_country_flag = YEM_civil_war
		custom_effect_tooltip = YEM_civil_war_tt
		if = {
			limit = {
				has_country_flag = YEM_arms_stockpile
			}
			add_equipment_to_stockpile = {
				type = infantry_equipment
				amount = 2000
			}
		}
		subtract_20_islam = yes
		hidden_effect = {
			set_capital = 814
			start_civil_war = {
				ruling_party = democratic
				ideology = neutrality
				size = 0.5
				capital = 293
				states = { 
					293 815
				}	
			    keep_unit_leaders = {
			    	8003
			    }
			}
			create_country_leader = {
				name = "Abdullah as-Sallal"
				desc = "POLITICS_SHABAB_AL-AMR_DESC"
				picture = "Portrait_YEM_Abdullah_as_Sallal.tga"
				expire = "1965.1.1"
				ideology = conservatism
				traits = {}
			}
			random_army_leader = { 
				limit = {
					has_id = 8001
				}
				retire = yes
			} 
			random_army_leader = { 
				limit = {
					has_id = 8002
				}
				retire = yes
			} 
			random_other_country = {
				limit = {
					original_tag = YEM
					has_government = neutrality
				}
				load_oob = "YEM_civil_war_neutrality"

				create_field_marshal = {
					name = "Yahya Muhammad Hamid ed-Din"
					picture = "portrait_yem_imam_yaha.dds"
					traits = {
						offensive_doctrine
					}
					skill = 1
					attack_skill = 1
					defense_skill = 1
					planning_skill = 1
					logistics_skill = 1
				}

				create_corps_commander = {
					name = "Ahmad bin Yahya"
					picture = "YEM_Ahmad_bin_Yahya.tga"
					traits = {
						harsh_leader
					}
					skill = 1
					attack_skill = 2
					defense_skill = 2
					planning_skill = 1
					logistics_skill = 1
				}
			}
			set_politics = {
				ruling_party = democratic
				elections_allowed = yes
			}
			load_oob = "YEM_civil_war_democratic"
			if = {
				limit = {
					AND = {
						democratic > 0.05
						democratic < 0.33
					}
				}
				add_ideas = YEM_democraticboost
			}
			if = {
				limit = {
					AND = {
						democratic > 0.33
						democratic < 0.50
					}
				}
				add_ideas = YEM_democraticboost2
			}
			if = {
				limit = {
					AND = {
						democratic > 0.50
					}
				}
				add_ideas = YEM_democraticboost3
			}
			remove_ideas = YEM_family_struggle
		}
	}
}
