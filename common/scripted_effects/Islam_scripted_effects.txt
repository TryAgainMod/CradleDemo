add_5_islam = {
	custom_effect_tooltip = add_five_islam_tt
	hidden_effect = {
		add_to_variable = {
		    var = islam_progress_bar
		    value = 5
		}
		set_variable = { islam = islam_progress_bar }
		divide_variable = { islam = 1000 }
		clamp_variable = { var = islam min = 0 max = 0.1 }
		set_variable = { islam_effect_2 = islam }
		multiply_variable = {
			var = islam_effect_2
			value = 1.2
		}
		set_variable = { islam_effect_3 = islam }
		multiply_variable = {
			var = islam_effect_3
			value = -1.0
		}
	}
}
